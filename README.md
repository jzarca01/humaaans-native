# Humaaans-Native

Humaaans.com for React-Native

## Install

`npm install --save humaaans-native`

## Usage

```javascript
import { Sitting1, Standing2 } from 'humaaans-native';

const StandingExample = () => (
  <Standing2
    height={200}
    hairColor="#000000"
    skinColor="#FDA7DC"
    shoeColor="#0000EE"
    pantColor="#FF44EA"
    shirtColor="#FFAB11"
    coatColor="#FF0E11"
    hatColor="#FFFFEE"
  />
);

const SittingExample = () => (
  <Sitting1
    height={200}
    hairColor="#000000"
    skinColor="#FDA7DC"
    shoeColor="#0000EE"
    pantColor="#FF44EA"
    shirtColor="#FFAB11"
    coatColor="#FF0E11"
    objectColor="#FFA011"
    hatColor="#FFFFEE"
    wheelchairColor="#FFBBEE"
  />
);
```
