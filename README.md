# React-Native Humaaans

Humaaans.com for React-Native

## Install

`npm install --save react-native-humaaans`

### How to use

```javascript
import { Sitting1, Standing2 } from 'react-native-humaaans';

const StandingExample = () => (
  <Standing2
    height={200}
    hairColor="#000000"
    skinColor="#FDA7DC"
    shoeColor="#0000EE"
    pantColor="#FF44EA"
    sweaterColor="#FF0011"
  />
);

const SittingExample = () => (
  <Sitting1
    height={200}
    hairColor="#000000"
    skinColor="#FDA7DC"
    shoeColor="#0000EE"
    pantColor="#FF44EA"
    shirtColor="#FFAB11"
    coatColor="#FF0E11"
    objectColor="#FFA011"
  />
);
```
